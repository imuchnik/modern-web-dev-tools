If we tried to run gulp it would fail, so we need the following modifications to our templates task


     gulp.task('templates', function(){

    var data = {}; //handlebars takes two arguments

  var options = {
    batch : ['./src/templates/partials']
  };

  return gulp.src([
    'src/templates/**/*.hbs',
    '!src/templates/partials/**/*.hbs' //ignore the partials directory
    ])
    .pipe(handlebars(data, options)) //provide those 2 argument
        so templating engine can load it

    .pipe(rename(function (path) {
      path.extname = '.html'; // give the files an .html extention
    }))
    .pipe(gulp.dest('./'));
});
